## Usage
1. Install package [man-super-ui]
   ```sh
   npm i -D man-super-ui
   ```
2. Get Started
   ```vue
    <template>
        <div id="app">
            <s-button type="primary">点击报名</s-button>
        </div>
    </template>

    <script>
    import {Button} from 'man-super-ui'
    import 'man-super-ui/dist/super.css'
    export default {
        name: 'app',
        components: {
            's-button':Button
        }
    }
    </script>

   ```
